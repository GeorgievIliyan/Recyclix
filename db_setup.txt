CREATE OR REPLACE FUNCTION public.approve_bin(bin_id uuid, bin_code text, bin_lat double precision, bin_lon double precision, bin_tags jsonb)
 RETURNS boolean
 LANGUAGE plpgsql
 SECURITY DEFINER
AS $function$
DECLARE
  success BOOLEAN := false;
BEGIN
  UPDATE pending_bins 
  SET status = 'approved', updated_at = NOW()
  WHERE id = bin_id;
  
  INSERT INTO recycling_bins (code, lat, lon, tags, stats_today, created_at, updated_at)
  VALUES (
    COALESCE(bin_code, bin_id::TEXT),
    bin_lat,
    bin_lon,
    bin_tags,
    '{}'::jsonb,
    COALESCE((SELECT created_at FROM pending_bins WHERE id = bin_id), NOW()),
    NOW()
  );
  
  success := true;
  RETURN success;
EXCEPTION WHEN OTHERS THEN
  RETURN false;
END;
$function$;

CREATE OR REPLACE FUNCTION public.approve_suggestion(suggestion_id uuid, reviewer_id uuid)
 RETURNS boolean
 LANGUAGE plpgsql
 SECURITY DEFINER
AS $function$
DECLARE
  suggestion_record edit_suggestions%ROWTYPE;
  bin_record recycling_bins%ROWTYPE;
  updated_tags JSONB;
  success BOOLEAN := false;
BEGIN
  SELECT * INTO suggestion_record FROM edit_suggestions WHERE id = suggestion_id;
  
  BEGIN
    SELECT * INTO bin_record FROM recycling_bins WHERE id::TEXT = suggestion_record.bin_id;
  EXCEPTION WHEN NO_DATA_FOUND THEN
    SELECT * INTO bin_record FROM recycling_bins WHERE code = suggestion_record.bin_id;
  END;
  
  IF bin_record.id IS NULL THEN
    RETURN false;
  END IF;
  
  updated_tags = bin_record.tags;
  
  IF suggestion_record.name IS NOT NULL THEN
    updated_tags = jsonb_set(updated_tags, '{name}', to_jsonb(suggestion_record.name));
  END IF;
  
  IF suggestion_record.opening_hours IS NOT NULL THEN
    updated_tags = jsonb_set(updated_tags, '{opening_hours}', to_jsonb(suggestion_record.opening_hours));
  END IF;
  
  IF suggestion_record.materials IS NOT NULL THEN
    updated_tags = jsonb_set(updated_tags, '{recycling_type}', to_jsonb(suggestion_record.materials));
  END IF;
  
  IF suggestion_record.field_name IS NOT NULL AND suggestion_record.new_value IS NOT NULL THEN
    updated_tags = jsonb_set(updated_tags, to_jsonb(suggestion_record.field_name), suggestion_record.new_value);
  END IF;
  
  UPDATE recycling_bins 
  SET tags = updated_tags, updated_at = NOW()
  WHERE id = bin_record.id;
  
  UPDATE edit_suggestions 
  SET 
    status = 'approved',
    reviewed_by = reviewer_id,
    reviewed_at = NOW(),
    updated_at = NOW()
  WHERE id = suggestion_id;
  
  success := true;
  RETURN success;
EXCEPTION WHEN OTHERS THEN
  RETURN false;
END;
$function$;

CREATE OR REPLACE FUNCTION public.check_admin_status(user_id uuid)
 RETURNS boolean
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
DECLARE
  user_role TEXT;
  is_admin_profile BOOLEAN;
BEGIN
  SELECT raw_user_meta_data->>'role' INTO user_role
  FROM auth.users
  WHERE id = user_id;
  
  IF user_role = 'service_role' THEN
    RETURN true;
  END IF;
  
  SELECT COALESCE(p.is_admin, false) INTO is_admin_profile
  FROM profiles p
  WHERE p.id = user_id;
  
  RETURN is_admin_profile;
EXCEPTION
  WHEN OTHERS THEN
    RETURN false;
END;
$function$;

CREATE OR REPLACE FUNCTION public.check_and_award_weight_badges(p_user_id uuid)
 RETURNS void
 LANGUAGE plpgsql
AS $function$
declare
  total_kg numeric;
begin
  select coalesce(sum(co2_saved), 0)
  into total_kg
  from recycling_events
  where user_id = p_user_id;

  if total_kg >= 10 then
    update user_profiles
    set badges = badges || '[1]'::jsonb
    where id = p_user_id
      and not (badges @> '[1]'::jsonb);
  end if;

  if total_kg >= 50 then
    update user_profiles
    set badges = badges || '[2]'::jsonb
    where id = p_user_id
      and not (badges @> '[2]'::jsonb);
  end if;

  if total_kg >= 250 then
    update user_profiles
    set badges = badges || '[3]'::jsonb
    where id = p_user_id
      and not (badges @> '[3]'::jsonb);
  end if;
end;
$function$;

CREATE OR REPLACE FUNCTION public.debug_admin_check(user_id uuid)
 RETURNS jsonb
 LANGUAGE plpgsql
 SECURITY DEFINER
AS $function$
DECLARE
  result JSONB;
  auth_user RECORD;
  profile_data RECORD;
  user_role TEXT;
BEGIN
  SELECT id, email, raw_user_meta_data, role INTO auth_user
  FROM auth.users
  WHERE id = user_id;
  
  user_role := auth_user.raw_user_meta_data->>'role';
  
  SELECT * INTO profile_data FROM profiles WHERE id = user_id;
  
  result := jsonb_build_object(
    'user_id', user_id,
    'auth_user_exists', auth_user.id IS NOT NULL,
    'auth_role', auth_user.role,
    'metadata_role', user_role,
    'raw_user_meta_data', auth_user.raw_user_meta_data,
    'profile_exists', profile_data.id IS NOT NULL,
    'profile_is_admin', COALESCE(profile_data.is_admin, false),
    'profile_data', to_jsonb(profile_data)
  );
  
  RETURN result;
END;
$function$;

CREATE OR REPLACE FUNCTION public.get_edit_suggestions()
 RETURNS TABLE(id uuid, bin_id text, name text, opening_hours text, materials text, notes text, status text, user_id uuid, created_at timestamp with time zone, reviewed_by uuid, reviewed_at timestamp with time zone, review_notes text, field_name text, old_value jsonb, new_value jsonb, reason text, user_email text, bin_code text, bin_lat double precision, bin_lon double precision, bin_tags jsonb)
 LANGUAGE plpgsql
 SECURITY DEFINER
AS $function$

  RETURN QUERY

    es.id, es.bin_id, es.name, es.opening_hours, es.materials, es.notes,

  LEFT JOIN recycling_bins rb ON rb.code = es.bin_id
  WHERE es.status = 'pending'
  ORDER BY es.created_at DESC;

$function$;

CREATE OR REPLACE FUNCTION public.get_reports()
 RETURNS TABLE(id uuid, bin_id text, user_id uuid, type text, title text, description text, resolved boolean, created_at timestamp with time zone, photo_url text, user_email text, user_username text, bin_code text, bin_lat double precision, bin_lon double precision, images jsonb)
 LANGUAGE plpgsql
 SECURITY DEFINER
AS $function$
BEGIN
  RETURN QUERY
  SELECT 
    r.id, r.bin_id, r.user_id, r.type::TEXT, r.title, r.description,
    r.resolved, r.created_at, r.photo_url,
    COALESCE(p.email, 'Анонимен') as user_email,
    COALESCE(p.username, p.email, 'Анонимен') as user_username,
    rb.code as bin_code,
    rb.lat as bin_lat,
    rb.lon as bin_lon,
    COALESCE(
      (SELECT jsonb_agg(jsonb_build_object(
        'id', rp.id,
        'photo_url', rp.photo_url,
        'created_at', rp.created_at,
        'user_id', rp.user_id
      ))
      FROM report_photos rp
      WHERE rp.report_id = r.id),
      '[]'::jsonb
    ) as images
  FROM reports r
  LEFT JOIN profiles p ON r.user_id = p.id
  LEFT JOIN recycling_bins rb ON rb.id::TEXT = r.bin_id
  ORDER BY r.created_at DESC;
END;
$function$;
