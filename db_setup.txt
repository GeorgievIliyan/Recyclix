CREATE OR REPLACE FUNCTION public.is_platform_admin()
RETURNS boolean LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
    RETURN EXISTS (
        SELECT 1 FROM public.user_profiles
        WHERE id = auth.uid() AND app_role = 'platform_admin'
    );
END;
$$;

CREATE OR REPLACE FUNCTION public.check_admin_status(p_user_id uuid)
RETURNS boolean LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE
    user_role TEXT;
    is_admin_profile BOOLEAN;
BEGIN
    SELECT raw_user_meta_data->>'role' INTO user_role FROM auth.users WHERE id = p_user_id;
    IF user_role = 'service_role' THEN RETURN true; END IF;
    SELECT (app_role IN ('platform_admin', 'manager')) INTO is_admin_profile FROM public.user_profiles WHERE id = p_user_id;
    RETURN COALESCE(is_admin_profile, false);
EXCEPTION WHEN OTHERS THEN
    RETURN false;
END;
$$;

CREATE OR REPLACE FUNCTION public.check_and_award_weight_badges(p_user_id uuid)
RETURNS void LANGUAGE plpgsql AS $$
DECLARE
    total_co2 numeric;
    glass_count integer;
    max_streak integer;
    prev_day date;
    curr_day date;
    curr_streak integer;
BEGIN
    SELECT COALESCE(SUM(co2_saved), 0) INTO total_co2 FROM public.recycling_events WHERE user_id = p_user_id;
    IF total_co2 >= 10 THEN UPDATE public.user_profiles SET badges = badges || '[1]'::jsonb WHERE id = p_user_id AND NOT (badges @> '[1]'::jsonb); END IF;
    IF total_co2 >= 50 THEN UPDATE public.user_profiles SET badges = badges || '[2]'::jsonb WHERE id = p_user_id AND NOT (badges @> '[2]'::jsonb); END IF;
    IF total_co2 >= 250 THEN UPDATE public.user_profiles SET badges = badges || '[3]'::jsonb WHERE id = p_user_id AND NOT (badges @> '[3]'::jsonb); END IF;
    
    SELECT COUNT(*) INTO glass_count FROM public.recycling_events WHERE user_id = p_user_id AND material = 'glass_metal';
    IF glass_count >= 15 THEN UPDATE public.user_profiles SET badges = badges || '[4]'::jsonb WHERE id = p_user_id AND NOT (badges @> '[4]'::jsonb); END IF;
    IF glass_count >= 100 THEN UPDATE public.user_profiles SET badges = badges || '[6]'::jsonb WHERE id = p_user_id AND NOT (badges @> '[6]'::jsonb); END IF;

    max_streak := 0; curr_streak := 1; prev_day := null;
    FOR curr_day IN SELECT DISTINCT date_trunc('day', created_at)::date FROM public.recycling_events WHERE user_id = p_user_id AND material = 'glass_metal' ORDER BY 1 LOOP
        IF prev_day IS NOT NULL AND curr_day = prev_day + 1 THEN curr_streak := curr_streak + 1;
        ELSIF prev_day IS NOT NULL THEN curr_streak := 1; END IF;
        IF curr_streak > max_streak THEN max_streak := curr_streak; END IF;
        prev_day := curr_day;
    END LOOP;
    IF max_streak >= 7 THEN UPDATE public.user_profiles SET badges = badges || '[5]'::jsonb WHERE id = p_user_id AND NOT (badges @> '[5]'::jsonb); END IF;
END;
$$;

CREATE OR REPLACE FUNCTION public.after_recycling_event()
RETURNS trigger LANGUAGE plpgsql AS $$
BEGIN
    PERFORM public.check_and_award_weight_badges(NEW.user_id);
    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION public.approve_bin(bin_id uuid, bin_code text, bin_lat double precision, bin_lon double precision, bin_tags jsonb)
RETURNS boolean LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
    UPDATE public.pending_bins SET status = 'approved', updated_at = NOW() WHERE id = bin_id;
    INSERT INTO public.recycling_bins (code, lat, lon, tags, stats_today, created_at, updated_at)
    VALUES (COALESCE(bin_code, bin_id::TEXT), bin_lat, bin_lon, bin_tags, '{}'::jsonb, COALESCE((SELECT created_at FROM public.pending_bins WHERE id = bin_id), NOW()), NOW());
    RETURN true;
EXCEPTION WHEN OTHERS THEN RETURN false;
END;
$$;

CREATE OR REPLACE FUNCTION public.approve_suggestion(suggestion_id uuid, reviewer_id uuid)
RETURNS boolean LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE
    suggestion_record public.edit_suggestions%ROWTYPE;
    bin_record public.recycling_bins%ROWTYPE;
    updated_tags JSONB;
BEGIN
    SELECT * INTO suggestion_record FROM public.edit_suggestions WHERE id = suggestion_id;
    BEGIN
        SELECT * INTO bin_record FROM public.recycling_bins WHERE id::TEXT = suggestion_record.bin_id;
    EXCEPTION WHEN NO_DATA_FOUND THEN
        SELECT * INTO bin_record FROM public.recycling_bins WHERE code = suggestion_record.bin_id;
    END;
    IF bin_record.id IS NULL THEN RETURN false; END IF;
    
    updated_tags := bin_record.tags;
    IF suggestion_record.name IS NOT NULL THEN updated_tags := jsonb_set(updated_tags, '{name}', to_jsonb(suggestion_record.name)); END IF;
    IF suggestion_record.opening_hours IS NOT NULL THEN updated_tags := jsonb_set(updated_tags, '{opening_hours}', to_jsonb(suggestion_record.opening_hours)); END IF;
    IF suggestion_record.materials IS NOT NULL THEN updated_tags := jsonb_set(updated_tags, '{recycling_type}', to_jsonb(suggestion_record.materials)); END IF;
    IF suggestion_record.field_name IS NOT NULL AND suggestion_record.new_value IS NOT NULL THEN updated_tags := jsonb_set(updated_tags, to_jsonb(suggestion_record.field_name), suggestion_record.new_value); END IF;
    
    UPDATE public.recycling_bins SET tags = updated_tags, updated_at = NOW() WHERE id = bin_record.id;
    UPDATE public.edit_suggestions SET status = 'approved', reviewed_by = reviewer_id, reviewed_at = NOW(), updated_at = NOW() WHERE id = suggestion_id;
    RETURN true;
EXCEPTION WHEN OTHERS THEN RETURN false;
END;
$$;

CREATE OR REPLACE FUNCTION public.create_user_profile(p_user uuid)

        INSERT INTO public.user_profiles (id, xp, level, trust_score, streak, created_at, updated_at) VALUES (p_user, 0, 1, 1.0, 0, now(), now());

END;
$$;

CREATE OR REPLACE FUNCTION public.debug_admin_check(p_user_id uuid)
RETURNS jsonb LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE
    result JSONB;
    auth_user RECORD;
    profile_data RECORD;
BEGIN
    SELECT id, email, raw_user_meta_data, role INTO auth_user FROM auth.users WHERE id = p_user_id;
    SELECT * INTO profile_data FROM public.user_profiles WHERE id = p_user_id;
    result := jsonb_build_object(
        'user_id', p_user_id, 'auth_user_exists', auth_user.id IS NOT NULL,
        'auth_role', auth_user.role, 'metadata_role', auth_user.raw_user_meta_data->>'role',
        'raw_user_meta_data', auth_user.raw_user_meta_data, 'profile_exists', profile_data.id IS NOT NULL,
        'profile_app_role', COALESCE(profile_data.app_role, 'user'), 'profile_data', to_jsonb(profile_data)
    );
    RETURN result;
END;
$$;

CREATE OR REPLACE FUNCTION public.get_edit_suggestions()
RETURNS TABLE(id uuid, bin_id text, name text, opening_hours text, materials text, notes text, status text, user_id uuid, created_at timestamp with time zone, reviewed_by uuid, reviewed_at timestamp with time zone, review_notes text, field_name text, old_value jsonb, new_value jsonb, reason text, user_email text, bin_code text, bin_lat double precision, bin_lon double precision, bin_tags jsonb)
LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
    RETURN QUERY SELECT es.id, es.bin_id, es.name, es.opening_hours, es.materials, es.notes, es.status::TEXT, es.user_id, es.created_at, es.reviewed_by, es.reviewed_at, es.review_notes, es.field_name, es.old_value, es.new_value, es.reason, COALESCE(p.username, 'Анонимен') AS user_email, rb.code AS bin_code, rb.lat AS bin_lat, rb.lon AS bin_lon, rb.tags AS bin_tags
    FROM public.edit_suggestions es LEFT JOIN public.user_profiles p ON es.user_id = p.id LEFT JOIN public.recycling_bins rb ON rb.code = es.bin_id
    WHERE es.status = 'pending' ORDER BY es.created_at DESC;
END;
$$;

CREATE OR REPLACE FUNCTION public.get_pending_bins()
RETURNS TABLE(id uuid, code text, lat double precision, lon double precision, created_at timestamp with time zone, user_id uuid, tags jsonb, image_url text, status text, user_email text, user_username text)

CREATE OR REPLACE FUNCTION public.get_reports()
RETURNS TABLE(id uuid, bin_id text, user_id uuid, type text, title text, description text, resolved boolean, created_at timestamp with time zone, photo_url text, user_email text, user_username text, bin_code text, bin_lat double precision, bin_lon double precision, images jsonb)
LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
    RETURN QUERY SELECT r.id, r.bin_id::TEXT, r.user_id, r.type::TEXT, r.title, r.description, r.resolved, r.created_at, r.photo_url, COALESCE(p.username, 'Анонимен') AS user_email, COALESCE(p.username, 'Анонимен') AS user_username, rb.code AS bin_code, rb.lat AS bin_lat, rb.lon AS bin_lon, COALESCE((SELECT jsonb_agg(jsonb_build_object('id', rp.id, 'photo_url', rp.photo_url, 'created_at', rp.created_at, 'user_id', rp.user_id)) FROM public.report_photos rp WHERE rp.report_id = r.id), '[]'::jsonb) AS images
    FROM public.reports r LEFT JOIN public.user_profiles p ON r.user_id = p.id LEFT JOIN public.recycling_bins rb ON rb.id = r.bin_id ORDER BY r.created_at DESC;
END;
$$;

CREATE OR REPLACE FUNCTION public.handle_password_change()
RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
    IF NEW.encrypted_password <> OLD.encrypted_password THEN
        UPDATE public.user_profiles SET password_changed_at = now() WHERE id = NEW.id;
    END IF;
    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.user_profiles (id, username, app_role)
  VALUES (new.id, new.raw_user_meta_data->>'username', COALESCE(new.raw_user_meta_data->>'role', 'user'));
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE OR REPLACE TRIGGER on_auth_user_created AFTER INSERT ON auth.users FOR EACH ROW EXECUTE PROCEDURE public.handle_new_user();
CREATE OR REPLACE TRIGGER on_recycling_event AFTER INSERT ON public.recycling_events FOR EACH ROW EXECUTE PROCEDURE public.after_recycling_event();
